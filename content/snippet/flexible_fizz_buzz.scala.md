```scala
def flexibleFizzBuzz(start: Int, end: Int) (callback: String => Unit): Unit = {
    for (i <- Range(start, end)) {
      var output: String = ""
      if (i % 3 == 0) {
        output += "Fizz"
      }
      if (i % 5 == 0) {
        output += "Buzz"
      }
      if (output.isEmpty) {
        output = s"${i}"
      }
      callback(output)
    }
}

flexibleFizzBuzz(start = 1, end = 50) {
  i => println(i)
}
```
